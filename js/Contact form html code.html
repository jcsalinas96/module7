    // Accessibility-friendly client-side validation & behavior
    (function(){
      const form = document.getElementById('contactForm');
      const nameInput = document.getElementById('name');
      const emailInput = document.getElementById('email');
      const phoneInput = document.getElementById('phone');
      const textsCheckbox = document.getElementById('texts');
      const privacyCheckbox = document.getElementById('privacy');

      const nameError = document.getElementById('nameError');
      const emailError = document.getElementById('emailError');
      const phoneError = document.getElementById('phoneError');
      const privacyError = document.getElementById('privacyError');

      // When texts checkbox is checked, make phone required
      textsCheckbox.addEventListener('change', () => {
        if(textsCheckbox.checked){
          phoneInput.required = true;
          phoneInput.setAttribute('aria-required', 'true');
        } else {
          phoneInput.required = false;
          phoneInput.removeAttribute('aria-required');
          phoneError.hidden = true;
        }
      });

      // Basic email regex (not exhaustive) for client-side hinting
      const emailRe = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      const phoneRe = /^\+?[\d\-\s().]{7,20}$/;

      form.addEventListener('submit', function(e){
        let ok = true;

        // Name
        if(!nameInput.value.trim()){
          nameError.hidden = false;
          ok = false;
        } else { nameError.hidden = true; }

        // Email
        if(!emailRe.test(emailInput.value.trim())){
          emailError.hidden = false;
          ok = false;
        } else { emailError.hidden = true; }

        // Phone: if required (texts checked) then validate pattern
        if(phoneInput.required){
          if(!phoneRe.test(phoneInput.value.trim())){
            phoneError.hidden = false;
            ok = false;
          } else { phoneError.hidden = true; }
        }

        // Privacy consent
        if(!privacyCheckbox.checked){
          privacyError.hidden = false;
          ok = false;
        } else { privacyError.hidden = true; }

        if(!ok){
          e.preventDefault();
          // focus the first invalid field for keyboard users
          const firstError = document.querySelector('.error:not([hidden])');
          if(firstError){
            const field = firstError.previousElementSibling;
            if(field && typeof field.focus === 'function') field.focus();
          }
        } else {
          // Optionally: show a small "submitting..." state or disable the submit button
          document.getElementById('submitBtn').disabled = true;
        }
      });

      // Hide errors on input
      [nameInput, emailInput, phoneInput, privacyCheckbox].forEach(el => {
        el.addEventListener('input', () => {
          const err = document.getElementById(el.id + 'Error');
          if(err) err.hidden = true;
        });
        el.addEventListener('change', () => {
          const err = document.getElementById(el.id + 'Error');
          if(err) err.hidden = true;
        });
      });

    })();
  </script>
